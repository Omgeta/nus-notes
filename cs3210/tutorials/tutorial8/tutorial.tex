int* process(int* images, int N) {
  int* results = new int[N][320][320];

  #pragma omp parallel for schedule(static, 2)
  for (int im = 0; im < N; im++) {
    for (int y = 0; y < 320; y++) {
      for (int x = 0; x < 320; x++) {
        int dxs[] = {-1, 0, 1};
        int dys[] = {-1, 0, 1};
        int sum = 0;

        bool valid = true;
        for (int dx : dxs) {
          for (int dy : dys) {
            int nx = x + dx;
            int ny = y + dy;
            if (nx >= 0 && ny >= 0 && nx < 320 && ny < 320)
              sum += images[im][nx][ny];
            else {
              valid = false;
            }
          }
        }
        
        results[im][x][y] = valid ? sum / 9 : images[im][x][y];
      }
    }
  }

  return results;
}
        \end{lstlisting}

        \item $= (\Pr(N=1) + \Pr(N=2) + \dots) + (\Pr(N=2) + \Pr(N=3) + \dots) + \dots$\hfill(Rearranging)
        \item $= \Pr[N\geq 1] + \Pr[N\geq 2] + \dots$\hfill(Additivity)
        \item $= \sum_{n>0}\Pr[N\geq n] \qed$\hfill(Summation)
        \end{enumproof}

      \item Derivation:
        \begin{enumproof}
        \item $\Pr[X \geq a] = \E[\1{X \geq a}]$\hfill(Indicator Law)
        \item $\leq \E[X / a]$\hfill(Using $\1{x \geq a} \leq x / a$ for $x\geq 0$)
        \item $= \E[X]/a \qed$\hfill(Scaling Law)

    \begin{align*}
      H(X) &= H_2(\Pr[X=1])\\ 
           &= H_2(\frac{1}{n}p_1 + \dots + \frac{1}{n}p_n)\tag*{(since $x_i$ chosen w.p $\frac{1}{n}$)}\\ 
           &= H_2(\frac{1}{n} \sum p_j)

    and also:
    \begin{align*}
      H(X|J) &= \sum_j \Pr[J = j]H(X|J=j)\\ 
             &= \frac{1}{n}\sum_jH_2(\Pr[X=1|J=j])\\ 
             &= \frac{1}{n}\sum_j H_2(p_j)
    \end{align*}
    Therefore, by conditioning reduces entropy we have:
    \begin{align*}
      H(X|J) &\leq H(X)\\
      \frac{1}{n}\sum_j H_2(p_j) &\leq H_2(\frac{1}{n}\sum_j p_j)
    \end{align*}

        \begin{align*}
          H(X) &= \sum_{a_i}P_X(a_i)\log(\frac{1}{P_X(a_i)})\\ 
               &= \sum^{n-1}_{i=1}P_X(a_i)\log(\frac{1}{P_X(a_i)}) + P_X(a_n)\log(\frac{1}{P_X(a_n)})\\ 
               &= \sum^{n-1}_{i=1}2^{-i}\log(\frac{1}{2^{-i}}) + 2^{-n+1}\log(\frac{1}{2^{-n+1}})\\ 
               &= \sum^{n-1}_{i=1}2^{-i}i + 2^{-n+1}(n-1)\\ 
               &= 2^{-n+1}(2^n-n-1)+2^{-n+1}(n-1)\\ 
               &= 2(1-2^{-n+1})
        \end{align*}

    \begin{align*}
      H(Y|X) &= \sum_{x\in\mcX}P_X(x)H(Y|X=x)\\ 
             &= \sum_{x\in\mcX}P_X(x)H_2(\delta)\\ 
             &= H_2(\delta)
    \end{align*}
    Then the capacity is:
    \begin{align*}
      C &= \max_{P_X}I(X;Y)\\ 
        &= \max_{P_X}H(Y) - H(Y|X)\\ 
        &= 2 - H_2(\delta)
    \end{align*}

                       &= I(X;Y_1) + I(X;Y_2) - I(Y_1;Y_2)\\

        \end{align*}

        \begin{align*}
          C' &= \max_{P_X}I(X;Y_1,Y_2)\\

    \end{align*}
    Consider,

      H(Y) &= \sum_{y\in\mcY} P_Y(y)\frac{1}{P_Y(y)}\\ 
           &= 2p\log_2 \frac{1}{p} + (1-2p)\log_2 \frac{1}{1-2p}\\ 
      H(Y|X) &= \sum_{x\in\mcX}P_X(x)H(Y|X=x)\\

             &= 2pH_2(\frac{1}{3})
    \end{align*}
    Capacity of the channel is:
    \begin{align*}
      C &= \max_{P_X}I(X;Y)\\
        &= \max_{P_X}H(Y) - H(Y|X)\\ 
        &= \max_p \left[2p\log_2 \frac{1}{p} + (1-2p)\log_2 \frac{1}{1-2p} -2pH_2(\frac{1}{3})\right]
    \end{align*}
    Differentiating the right side, we find maximum at $p=\frac{1}{2 + 2^{H_2(\frac{1}{3})}}$

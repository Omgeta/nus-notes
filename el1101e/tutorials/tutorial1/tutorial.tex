  \item Singaporean;\\ 
    Deletion of postvocalic /r/: \textipa{[titS@r]} $\rightarrow$ \textipa{[titS@]}\\
    Final consonant devoicing:  \textipa{[bIliz]} $\rightarrow$ \textipa{[bilis]}; \textipa{[kOld]} $\rightarrow$ \textipa{[kOlt]}\\
    Merger of vowels: \textipa{[lIt@l]} $\rightarrow$ \textipa{[lit@l]}; \textipa{[bIli]} $\rightarrow$ \textipa{[bili]}; \textipa{[d\ae d]} $\rightarrow$ \textipa{[dEd]}; \textipa{[deI]} $\rightarrow$ \textipa{[dei]} 

  \item US (GA):\\ 
    Unmerged vowels: \textipa{[I]}\\ 
    Diphthongs: \textipa{[eI]} in \textipa{[t@deI]}\\ 
    Rhotic: \textipa{[sA\textturnr i]}, \textipa{[g@\textturnr l]}, \textipa{[f\textturnr Vm]}\\
    LOT vowel: \textipa{[kapid]} instead of \textipa{[k\textturna pid]} (RP)

  \item UK (RP):\\ 
    Unmerged vowels: \textipa{[I]}\\ 
    LOT vowel: \textipa{[k\textturna pi]}, \textipa{[fr\textturna m]}\\
    COAT diphthong: \textipa{[@U]} in \textipa{[n@U]} 

  \item 
    \begin{enumerate}[(\alph*)]
      \item True Diphthongs: \textipa{[oU]}, \textipa{[eI]}, \textipa{[aI]}
      \item Mid Vowels: \textipa{[E]}, \textipa{[@]}
      \item Lax Vowels: \textipa{[E]}, \textipa{[@]}, \textipa{[I]}, \textipa{[\ae]}, \textipa{[2]} 
      \item Voiced Plosives: \textipa{[b]}, \textipa{[d]}
      \item Voiceless Fricatives: \textipa{[T]}, \textipa{[s]}, \textipa{[h]}
    \end{enumerate}

  \item 
    \begin{enumerate}[(\alph*)]
      \item \textipa{[T]}: inter-dental voiceless fricative\\ 
        \textipa{[D]}: inter-dental voiced fricative

      \item \textipa{[T]} $\rightarrow$ \textipa{[f]}, \textipa{[D]} $\rightarrow$ \textipa{[v]}: change in place (dental $\rightarrow$ labiodental)\\

      \item \textipa{[T]} $\rightarrow$ \textipa{[t]}, \textipa{[D]} $\rightarrow$ \textipa{[d]}: change in place (dental $\rightarrow$ alveolar) and manner (fricative $\rightarrow$ plosive)\\

      \item \textipa{[T]} $\rightarrow$ \textipa{[s]}, \textipa{[D]} $\rightarrow$ \textipa{[z]}: change in place (dental $\rightarrow$ alveolar)\\
    \end{enumerate}
